function App() {
	const [initialCount, setInitialCount] = useState({count:0, name: 'bob'});
	const [previousCount, setPreviousCount] = useState({count:0, name: 'bob'});
  const handleClick = () => {
    setInitialCount({
      ...initialCount,
      count: initialCount.count + 1,
    });
    setPreviousCount({
      ...initialCount,
      count: initialCount.count,
    });
  }
	return (
		<div className='App'>
			<ul>
				{/* {data.map(({id, name}) => <li id={id}>{name} {counter++}</li>)} */}
				{/* <inputintialCount
					value={initialCount}
					placeholder='HOLA'
					onChange={logger}
				/> */}
        <h1>
          Current count: {initialCount.count}
        </h1>
        <h1>
          Previous Count: {previousCount.count}
        </h1>

        <button onClick={handleClick}>Click Me plus 1</button>
			</ul>
		</div>
	);
}

export default App;
